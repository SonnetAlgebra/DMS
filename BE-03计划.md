import pandas as pd
from typing import List, Dict, Tuple
import io

class CSVAdapter:
    def parse(self, file_content: bytes) -> Tuple[List[str], List[Dict]]:
        """
        解析CSV文件
        返回: (指标名称列表, 数据点列表)
        数据点格式: {metric_name, timestamp, value}
        """
        # 1. 读取CSV（UTF-8编码，处理BOM）
        df = pd.read_csv(io.BytesIO(file_content), encoding='utf-8-sig')
        
        # 2. 识别时间戳列（支持timestamp/time/日期，大小写不敏感）
        timestamp_col = self._find_timestamp_column(df.columns)
        
        # 3. 识别数值列（除时间戳外的所有列）
        metric_cols = [c for c in df.columns if c != timestamp_col]
        
        # 4. 转换为标准格式，保留NULL
        data_points = []
        for col in metric_cols:
            for _, row in df.iterrows():
                value = row[col]
                # 保留NULL：pandas NaN -> None
                if pd.isna(value):
                    value = None
                data_points.append({
                    "metric_name": col,
                    "timestamp": row[timestamp_col],
                    "value": value
                })
        
        return metric_cols, data_points
    
    def _find_timestamp_column(self, columns) -> str:
        candidates = ['timestamp', 'time', '日期', '时间', 'ts', 'datetime']
        for col in columns:
            if col.lower() in candidates:
                return col
        raise ValueError("未找到时间戳列")
步骤3：实现上传路由（routers/upload.py）

复制
from fastapi import APIRouter, UploadFile, File
from app.adapters.csv_adapter import CSVAdapter
from app.schemas.upload import UploadResponse
import sqlite3
from datetime import datetime

router = APIRouter()
adapter = CSVAdapter()
DB_PATH = "./monitor.db"

@router.post("/upload", response_model=UploadResponse)
async def upload_csv(file: UploadFile = File(...)):
    # 1. 读取文件内容
    content = await file.read()
    
    # 2. 解析CSV
    metric_names, data_points = adapter.parse(content)
    
    # 3. 原生SQLite操作
    conn = sqlite3.connect(DB_PATH)
    cursor = conn.cursor()
    
    inserted_count = 0
    imported_metrics = []
    
    try:
        for metric_name in metric_names:
            # 检查/创建指标
            cursor.execute("SELECT id FROM metrics WHERE name = ?", (metric_name,))
            row = cursor.fetchone()
            if row:
                metric_id = row[0]
            else:
                cursor.execute(
                    "INSERT INTO metrics (name, description) VALUES (?, ?)",
                    (metric_name, f"指标: {metric_name}")
                )
                metric_id = cursor.lastrowid
                imported_metrics.append({"id": metric_id, "name": metric_name})
            
            # 批量插入时序数据（保留NULL）
            metric_data = [d for d in data_points if d["metric_name"] == metric_name]
            for d in metric_data:
                cursor.execute(
                    "INSERT OR IGNORE INTO time_series (metric_id, timestamp, value) VALUES (?, ?, ?)",
                    (metric_id, d["timestamp"], d["value"])
                )
                inserted_count += cursor.rowcount
        
        conn.commit()
    finally:
        conn.close()
    
    return UploadResponse(
        success=True,
        inserted=inserted_count,
        metrics=imported_metrics
    )

关键约束
NULL值直接插入，不填充默认值
使用INSERT OR IGNORE避免重复数据报错
原生SQLite操作，不使用ORM
UTF-8编码，自动处理BOM头
时间戳列名大小写不敏感识别
验收标准
[ ] POST /api/v1/data/upload 接收CSV文件返回200
[ ] 响应包含success: true、inserted计数、metrics列表
[ ] CSV中NULL值（空字符串或缺失）入库后为NULL
[ ] 重复上传同一文件不报错（OR IGNORE生效）
[ ] 支持中文列名和UTF-8编码
